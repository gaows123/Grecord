107 virtual inline unsigned char computeCost(double distance) const
108 {
109 unsigned char cost = 0;
110 if (distance == 0)
111 cost = LETHAL_OBSTACLE;
112 else if (distance * resolution_ <= inscribed_radius_)
113 cost = INSCRIBED_INFLATED_OBSTACLE;
114 else
115 {
116 // make sure cost falls off by Euclidean distance
117 double euclidean_distance = distance * resolution_;
118 double factor = exp(-1.0 * weight_ * (euclidean_distance - inscribed_radius_));
119 cost = (unsigned char)((INSCRIBED_INFLATED_OBSTACLE - 1) * factor);
120 }
121 return cost;
122 }
