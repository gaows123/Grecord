111 inline bool gridCoords(geometry_msgs::Point pt, unsigned int& gx, unsigned int& gy) const {
112 if(pt.x < origin_.x || pt.y < origin_.y){
113 gx = 0;
114 gy = 0;
115 return false;
116 }
117 gx = (int) ((pt.x - origin_.x)/resolution_);
118 gy = (int) ((pt.y - origin_.y)/resolution_);
119
120 if(gx >= width_ || gy >= height_){
121 gx = 0;
122 gy = 0;
123 return false;
124 }
125
126 return true;
127 }

140 upper_right.x = lower_left.x + resolution_;

186 inline unsigned int gridIndex(unsigned int gx, unsigned int gy) const {

206 inline double orient(const geometry_msgs::Point& a, const geometry_msgs::Point& b, const geometry_msgs::Point32& c){
207 double acx = a.x - c.x;
208 double bcx = b.x - c.x;
209 double acy = a.y - c.y;
210 double bcy = b.y - c.y;
211 return acx * bcy - acy * bcx;
212 }
