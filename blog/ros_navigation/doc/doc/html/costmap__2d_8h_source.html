317 template<typename data_type>
318 void copyMapRegion(data_type* source_map, unsigned int sm_lower_left_x, unsigned int sm_lower_left_y,
319 unsigned int sm_size_x, data_type* dest_map, unsigned int dm_lower_left_x,
320 unsigned int dm_lower_left_y, unsigned int dm_size_x, unsigned int region_size_x,
321 unsigned int region_size_y)
322 {
323 // we'll first need to compute the starting points for each map
324 data_type* sm_index = source_map + (sm_lower_left_y * sm_size_x + sm_lower_left_x);
325 data_type* dm_index = dest_map + (dm_lower_left_y * dm_size_x + dm_lower_left_x);
326
327 // now, we'll copy the source map into the destination map
328 for (unsigned int i = 0; i < region_size_y; ++i)
329 {
330 memcpy(dm_index, sm_index, region_size_x * sizeof(data_type));
331 sm_index += sm_size_x;
332 dm_index += dm_size_x;
333 }
334 }

362 template<class ActionType>
363 inline void raytraceLine(ActionType at, unsigned int x0, unsigned int y0, unsigned int x1, unsigned int y1,
364 unsigned int max_length = UINT_MAX)
365 {
366 int dx = x1 - x0;
367 int dy = y1 - y0;
368
369 unsigned int abs_dx = abs(dx);
370 unsigned int abs_dy = abs(dy);
371
372 // 标记x是前进一格还是后退一个单元格
373 int offset_dx = sign(dx);
374 int offset_dy = sign(dy) * size_x_;
375
376 unsigned int offset = y0 * size_x_ + x0;
377
378 // we need to chose how much to scale our dominant dimension, based on the maximum length of the line
379 double dist = hypot(dx, dy);
380 double scale = (dist == 0.0) ? 1.0 : std::min(1.0, max_length / dist);
381
382 // if x is dominant
383 if (abs_dx >= abs_dy)
384 {
385 int error_y = abs_dx / 2;
386 bresenham2D(at, abs_dx, abs_dy, error_y, offset_dx, offset_dy, offset, (unsigned int)(scale * abs_dx));
387 return;
388 }
389
390 // otherwise y is dominant
391 int error_x = abs_dy / 2;
392 bresenham2D(at, abs_dy, abs_dx, error_x, offset_dy, offset_dx, offset, (unsigned int)(scale * abs_dy));
393 }

400 template<class ActionType>
401 inline void bresenham2D(ActionType at, unsigned int abs_da, unsigned int abs_db, int error_b, int offset_a,
402 int offset_b, unsigned int offset, unsigned int max_length)
403 {
404 unsigned int end = std::min(max_length, abs_da);
405 for (unsigned int i = 0; i < end; ++i)
406 {
407 at(offset);
408 offset += offset_a;
409 /*
410 * 下面这个情况，error_b + abs_db就会超过abs_da，所以每次x+1后，y也会+1
411 * _
412 * _|
413 * _|
414 * _|
415 * _|
416 *
417 * 下面这个情况，error_b + n* abs_db才会超过abs_da,所以每次x+n后，y才会+1
418 * ___|
419 * ___|
420 * ___|
421 * ___|
422 */
423 // https://blog.csdn.net/TurboIan/article/details/86611713
424 error_b += abs_db; // 为了控制前进上升的斜率
425 if ((unsigned int)error_b >= abs_da)
426 {
427 offset += offset_b;
428 error_b -= abs_da;
429 }
430 }
431 at(offset);
432 }
