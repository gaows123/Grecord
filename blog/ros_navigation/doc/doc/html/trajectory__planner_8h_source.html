324 inline double computeNewXPosition(double xi, double vx, double vy, double theta, double dt){
325 return xi + (vx * cos(theta) + vy * cos(M_PI_2 + theta)) * dt;
326 }
327
337 inline double computeNewYPosition(double yi, double vx, double vy, double theta, double dt){
338 return yi + (vx * sin(theta) + vy * sin(M_PI_2 + theta)) * dt;
339 }
340
348 inline double computeNewThetaPosition(double thetai, double vth, double dt){
349 return thetai + vth * dt;
350 }

352 //compute velocity based on acceleration
361 inline double computeNewVelocity(double vg, double vi, double a_max, double dt){
362 if((vg - vi) >= 0) {
363 return std::min(vg, vi + a_max * dt);
364 }
365 return std::max(vg, vi - a_max * dt);
366 }
